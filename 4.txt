int main()
{
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	const int n = 5;
	int matrix1[n][n];
	int matrix2[n][n];
	int matrix_mult[n][n];

	cout << "Матриця №1:\n";
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			matrix1[i][j] = rand() % 10;
			matrix_mult[i][j] = 0;
			cout << matrix1[i][j] << "\t";
		}
		cout << "\n";
	}

	cout << "------------------------------------\nДруга матриця:\n";
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			matrix2[i][j] = rand() % 10;
			cout << matrix2[i][j] << "\t";
		}
		cout << "\n";
	}

	cout << "------------------------------------\nДобуток матриць:\n";
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			for (int k = 0; k < n; k++)
			{
				matrix_mult[i][j] += matrix1[i][k] * matrix2[k][j];
				
			}
			cout << matrix_mult[i][j] << "\t";
		}
		cout << "\n";
	}
}
