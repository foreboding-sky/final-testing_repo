{

	SetConsoleOutputCP(1251);
	double sum = 0, count = 0;
	double arr[20];
	cout << "Масив:\n";
	for (int i = 0; i < 20; i++)
	{
		arr[i] = rand() % 101 - 50;
		cout << "Arr[" << i + 1 << "] = " << arr[i] << endl;
	}
	for (auto number : arr)
	{
		if (number > 0)
		{
			sum += number;
			count++;
		}
	}
	cout << "\nСереднє арифметичне невід’ємних чисел: " << sum / count << endl;

}

		cout << "Arr[" << i + 1 << "]= " << arr[i] << endl;

#include <iostream>
#include <string>
#include "windows.h"
#include <cmath>
#include <array>
using namespace std;
int main()
{
	srand(time(NULL));
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	int arr[20];
	int mod[10];
	int max_el = 0;
	for (auto& elem : mod) {
		elem = 0;
	}
	cout << "array: ";
	for (int i = 0; i < 20; i++)
	{
		arr[i] = rand() % 10;
		cout << "Arr[" << i + 1 << "]= " << arr[i] << endl;
	}

	for (int i = 0; i < 20; i++) {
		mod[arr[i]]++;
	}
	max_el = mod[0];
	for (int i = 0; i < 10; i++) {
		if (max_el < mod[i]) max_el = i;
	}
	cout << "Мода масиву: " << max_el << endl;
}
